用hashmap和双向链表实现的LRU.

```java
class LRUCache {
    Map<Integer, DListNode> map;
    DListNode head;
    DListNode tail;
    int cap;
    public LRUCache(int capacity) {
        cap = capacity;
        map = new HashMap<>();
        head = new DListNode(0,0);
        tail = new DListNode(0,0);
        head.next = tail;
        tail.prev = head;
    }
    
    public int get(int key) {
        if (!map.containsKey(key)) {
            return -1;
        } 
        DListNode listNode = map.get(key);
        remove(listNode);
        addHead(listNode);
        return listNode.val;
    } 
    
    public void put(int key, int value) {
        if (map.containsKey(key)) {
            DListNode listNode = map.get(key);
            listNode.val = value;
            remove(listNode);
            addHead(listNode);
        } else {
            DListNode listNode = new DListNode(key, value);
            addHead(listNode);
            if (map.size() == cap) {
                map.remove(tail.prev.key);
                remove(tail.prev);
            }
            map.put(key, listNode);
        }
    }
    
    private void remove(DListNode node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }
    
    private void addHead(DListNode listNode) {
        DListNode tmp = head.next;
        head.next = listNode;
        listNode.next = tmp;
        tmp.prev = listNode;
        listNode.prev = head;
    }
}

class DListNode {
    int key;
    int val;
    DListNode next;
    DListNode prev;
    public DListNode(int k, int v) {
        key = k;
        val = v;
    }
}
```
